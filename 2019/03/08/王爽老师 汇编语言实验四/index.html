王爽老师 汇编语言实验四
2019-03-08
  1. 编程，向内存0:200023f依次传送数据063，程序中只能使用9条指令，包括“mov ax,4c00h”和”int 21h” 源代码：
    assume cs:codesg
    codesg segment
    mov ax,0020h
    mov ds,ax
    mov bx,0
    mov cx,64//0-63一共64个数 需要传值64次
    s:mov ds:[bx],bx将bx中的值传给基址为ds 偏移为bx的内存中。
    inc bx
    loop s
    mov ax,4c00h
    int 21h
    codesg ends
    end
